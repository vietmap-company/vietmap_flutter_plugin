Map<String, dynamic> toJson() {
  final Map<String, dynamic> data = <String, dynamic>{};
  data['display'] = display;
  data['name'] = name;
  data['hs_num'] = hsNum;
  data['street'] = street;
  data['address'] = address;
  data['city_id'] = cityId;
  data['city'] = city;
  data['district_id'] = districtId;
  data['district'] = district;
  data['ward_id'] = wardId;
  data['ward'] = ward;
  data['lat'] = lat;
  data['lng'] = lng;
  return data;
}
